"Do Work" in C# 


void signature(string action, string inOut, string place, string location)
{
    Console.WriteLine($"{action}, please sign {inOut} on
                    the {place} located by {location}.");
}

void work(string[] items)
{
    foreach (string item in items) {
        Console.WriteLine($"work on {item}");
    }
}

string[] workItems = {
    "assessments",
    "take break",
    "assessments",
};

signature("On arrival", "in", "podium", "door");
work(workItems);
signature("Before departure", "out", "podium", "door");

"Do Work" in Python 


def signature(action, inOut, place, location):
    print(f"{action}, please sign {inOut} " \
          f"on the {place} located by {location}.")

def work(items=[]):
    for item in items:
        print(f"work on {item}")

workItems = [
    "assessments",
    "take break",
    "assessments",
]
if __name__ == '__main__':
    signature("On arrival", "in", "podium", "door")
    work(workItems)
    signature("Before departure", "out", "podium", "door")
